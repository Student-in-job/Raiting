k float default 0, -- общее количество учебных курсов
--i7
ty float default 0, -- общее количество ппс, участвовавшего на тестировании
p7 float default 0, -- общее количество ппс, удовлетворительно сдавшего тесты по ин.языку

--i8

--i9
p9 float default 0, -- количество ппс, проводящего уч.занятия в учреждениях сспо
--i10
tsi float default 0, -- число студентов с положительной оценкой знаний по тестированию
tsti float default 0, -- число студентов, учавствовавших в тестировании
--i11
r1 float default 0, -- число положительных отзывов работодателей
r float default 0, -- число опрошенных выпускников
--i12

--i13

p1_3 float default 0, -- число студентов занявших призовые места на республиканских олимпиадах (по профилю)
p1_4 float default 0, -- число студентов занявших призовые места на международных олимпиадах (не по профилю)
p1_5 float default 0, -- число студентов занявших призовые места на республиканских олимпиадах (не по профилю)
--i14

--i15 
s7 float default 0, -- число научных статей в зарубежных журналах
s8 float default 0, -- число изданных монографий
s9 float default 0, -- число научных статей в республиканских научных журналах ВАК
--i16
d float default 0, -- количество утвержденных Президиумом ВАК РУз. в течение года докторских диссертаций 
z float default 0, -- количество сотрудников и профессоров-преподавателей, имеющих ученые степени
v float default 0, -- количество сотрудников, профессоров-преподавателей, имеющих ученые звания, не имеющих  ученой степени
--i17
t4 float default 0, -- финансовые средства, полученные за счет грантов и заказов зарубежных научных фондов и исследовательских центров
t5 float default 0, -- финансовые средства, полученные от иссле¬дований, проведенных по заказам соответствующих отраслей
t6 float default 0, -- финансовые средства, полученные от иссле¬дований, проведенных за счет государственных грантов
tL float default 0, -- финансовые средства, полученные от заключенных лицензионных соглашений
--i18
i float default 0, -- количество инновационных  проектов, выполняемых с участием (на базе)  соответствующих организаций
t7 float default 0, -- количество охранных документов (патентов) на объекты интеллектуальной собственности
t8 float default 0, -- количество внедренных в практику научных разработок, подтверждаемых актами внедрения соответствующих органов управления
ya float default 0, -- количество договоров вузов участвовавших на Ярмарке инновационных идей, технологий и проектов
kL float default 0, -- количество лицензионных соглашений, заключенных вузом по научно-технологическим разработкам ППС
--i19
a1 float default 0, -- количество лабораторных оборудований выпущенных в течении последних 5 лет
a2 float default 0, -- количество лабораторных оборудований выпущенные в течении последующих 5-10 лет
a3 float default 0, -- количество лабораторных оборудований выпущенные за последующие 10-20 лет
a4 float default 0, -- количество лабораторных оборудований выпущенные свыше 20 лет назад
a5 float default 0, -- количество лабораторных оборудований не пригодные к использованию
--i20
k2 float default 0, -- количество наименований литературы в наличии 
m float default 0, -- количества наименований литературы в электронном виде
L float default 0-- требуемое количество наименований основной литературы 
);

GO
